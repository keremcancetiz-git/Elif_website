// ─────────────────────────── SCROLL SPY ───────────────────────────
function showSection(id) {
    const target = document.getElementById('section-' + id);
    if (target) {
        const offsetPosition = target.getBoundingClientRect().top + window.pageYOffset - 100;
        window.scrollTo({ top: offsetPosition, behavior: 'smooth' });
    }
}

const sections = document.querySelectorAll('main > div[id^="section-"]');
const navButtons = {
    'section-work':  document.getElementById('nav-work'),
    'section-about': document.getElementById('nav-about')
};
const sectionObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            Object.values(navButtons).forEach(btn => { if (btn) btn.classList.remove('active'); });
            const btn = navButtons[entry.target.id];
            if (btn) btn.classList.add('active');
        }
    });
}, { rootMargin: '-50% 0px -50% 0px', threshold: 0 });
sections.forEach(s => sectionObserver.observe(s));

// ─────────────────────────── RENDER ───────────────────────────────
function observeReveal() {
    const ro = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('visible');
                ro.unobserve(entry.target);
            }
        });
    }, { threshold: 0.05 });
    setTimeout(() => {
        document.querySelectorAll('.project-card').forEach((card, idx) => {
            if (!card.classList.contains('reveal')) {
                card.classList.add('reveal');
                card.style.transitionDelay = `${idx * 70}ms`;
            }
        });
        document.querySelectorAll('.reveal').forEach(el => ro.observe(el));
    }, 10);
}

                <p style="font-size:15px; color:var(--muted); line-height:1.75; margin-bottom:32px; max-width:600px; font-weight:300;">${p.logicSummary}</p>

                <div onclick="openLogicZoom('${p.logicMap}')"
                    style="position:relative; border-radius:12px; overflow:hidden; border:1px solid var(--border); aspect-ratio:16/9; display:flex; align-items:center; justify-content:center; cursor:zoom-in; background:var(--cream); transition:border-color .2s;"
                    onmouseover="this.style.borderColor='var(--accent)'" onmouseout="this.style.borderColor='var(--border)'">
                    <img src="${p.logicMap}" style="position:absolute; inset:0; width:100%; height:100%; object-fit:cover; object-position:${p.logicPos || '50% 50%'}; opacity:0.25; transition:opacity .3s;"
                        onmouseover="this.style.opacity=0.45" onmouseout="this.style.opacity=0.25">
                    <div style="text-align:center; position:relative; z-index:1; padding:24px;">
                        <i data-lucide="zoom-in" style="width:32px; height:32px; color:var(--accent); margin:0 auto 12px; display:block;"></i>
                        <p style="font-weight:500; font-size:16px; color:var(--ink); margin-bottom:4px;">View Full Technical Schematic</p>
                        <p style="font-size:12px; color:var(--muted);">Pannable high-resolution logic map</p>
                    </div>
                </div>
            </section>

            <!-- Prev / Next -->
            <div style="display:flex; justify-content:space-between; align-items:center; padding-top:48px; border-top:1px solid var(--border);">
                ${prev ? `<button onclick="openModal('${prev.id}')" style="display:flex; align-items:center; gap:12px; text-align:left; cursor:pointer; border:none; background:none; transition:opacity .2s;" onmouseover="this.style.opacity=.6" onmouseout="this.style.opacity=1">
                    <i data-lucide="arrow-left" style="width:18px; height:18px; color:var(--muted);"></i>
                    <div>
                        <div class="label" style="margin-bottom:3px;">Previous</div>
                        <p style="font-family:'Cormorant Garamond',serif; font-size:20px; font-weight:600; color:var(--ink);">${prev.title}</p>
                    </div>
                </button>` : '<div></div>'}

                <button onclick="closeModal()" style="font-size:11px; font-weight:500; letter-spacing:0.1em; text-transform:uppercase; color:var(--muted); cursor:pointer; border:none; background:none; display:flex; align-items:center; gap:6px; transition:color .2s;"
                    onmouseover="this.style.color='var(--ink)'" onmouseout="this.style.color='var(--muted)'">
                    <i data-lucide="grid" style="width:14px; height:14px;"></i> All Projects
                </button>

                ${next ? `<button onclick="openModal('${next.id}')" style="display:flex; align-items:center; gap:12px; text-align:right; cursor:pointer; border:none; background:none; transition:opacity .2s;" onmouseover="this.style.opacity=.6" onmouseout="this.style.opacity=1">
                    <div>
                        <div class="label" style="margin-bottom:3px;">Next</div>
                        <p style="font-family:'Cormorant Garamond',serif; font-size:20px; font-weight:600; color:var(--ink);">${next.title}</p>
                    </div>
                    <i data-lucide="arrow-right" style="width:18px; height:18px; color:var(--muted);"></i>
                </button>` : '<div></div>'}
            </div>
        </div>
    `;

    modal.classList.remove('hidden');
    modal.classList.remove('animate-in');
    // Force reflow so slide-in animation re-triggers when navigating prev/next.
    void modal.offsetWidth;
    modal.classList.add('animate-in');
    document.body.classList.add('no-scroll');
    lucide.createIcons();
    updateStep(0, id);
}
